<script>

    const GIST_URL = 'https://gist.githubusercontent.com/charliemitchellstuff/e773c93b47513065379147179bfc48d5/raw/2fb29efdd1e6ac80a67dfdff64c177b893b15bac/complaints.txt';

   

    let allComplaints = [];

    let availableComplaints = [];

    let slots = [];

    function calculateGrid() {

        slots = [];

        const container = document.getElementById('container');

        const stepX = 270;

        const stepY = 175;

        const columns = Math.floor(container.offsetWidth / stepX);

        const rows = Math.floor(container.offsetHeight / stepY);

        const xOffset = (container.offsetWidth - (columns * stepX)) / 2;

        const yOffset = (container.offsetHeight - (rows * stepY)) / 2;

        for (let r = 0; r < rows; r++) {

            for (let c = 0; c < columns; c++) {

                slots.push({

                    x: c * stepX + xOffset + 15,

                    y: r * stepY + yOffset + 20,

                    inUse: false

                });

            }

        }

    }

    async function init() {

        try {

            const response = await fetch(GIST_URL);

            const text = await response.text();

            allComplaints = text.split('\n')

                .map(c => c.trim())

                .filter(c => c.length > 8);

           

            availableComplaints = [...allComplaints];

           

            if (allComplaints.length > 0) {

                calculateGrid();

                window.addEventListener('resize', calculateGrid);

                spawn();

                setInterval(spawn, 1300);

            }

        } catch (err) { console.error(err); }

    }

    function spawn() {

        const availableSlots = slots.filter(s => !s.inUse);

        if (availableSlots.length === 0 || availableComplaints.length === 0) return;

        const slot = availableSlots[Math.floor(Math.random() * availableSlots.length)];

        slot.inUse = true;

        const textIndex = Math.floor(Math.random() * availableComplaints.length);

        const selectedText = availableComplaints.splice(textIndex, 1)[0];

        const container = document.getElementById('container');

        const div = document.createElement('div');

        div.className = 'complaint';

        div.style.left = `${slot.x}px`;

        div.style.top = `${slot.y}px`;

        div.innerHTML = `<span class="label">Complaint</span><div class="text-content">${selectedText}</div>`;

       

        container.appendChild(div);

        setTimeout(() => div.classList.add('visible'), 50);

        const visibleDuration = 5000;

        setTimeout(() => { div.style.opacity = '0'; }, visibleDuration);

        setTimeout(() => {

            div.remove();

            slot.inUse = false;

            availableComplaints.push(selectedText);

        }, visibleDuration + 1000);

    }

    init();

</script>
